# 路由

/hosts/machinetype/

# 获取机器型号内容

查询方法：get

响应内容

字段名称 字段类型 含义
id int 自增id
name string 机器型号
cpu int cpu核数
memory int 内存大小(GB)
issuer int 云商id，AWS = 0，腾讯云 =1 ，华为云 = 2，ucloud=3
{
	"data": [        
        		{
                "id": 1,
                "name": "c5.24xlarge",
                "cpu": 96,
                "memory": 192,
                "issuer": 0
        }
    ],
    "code": 200,
    "message": "OK"
}
